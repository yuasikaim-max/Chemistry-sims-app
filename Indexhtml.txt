  <h1>Chemistry Simulation Prototype</h1>
  <p>Type a chemistry topic (e.g., "nitrogen cycle", "sulfur formation", "water cycle") and tap "Generate Simulation".</p>

  <input type="text" id="topicInput" placeholder="Enter topic here" />
  <button onclick="generateSimulation()">Generate Simulation</button>

  <div id="output"></div>

  <script>
    const simulations = {
      "nitrogen cycle": {
        text: "The nitrogen cycle is a natural process where nitrogen moves between the atmosphere, soil, and living organisms, essential for life.",
        image: "https://upload.wikimedia.org/wikipedia/commons/thumb/7/7e/Nitrogen_Cycle.svg/600px-Nitrogen_Cycle.svg.png"
      },
      "sulfur formation": {
        text: "Sulfur is formed through volcanic activity and decay of organic matter, and it plays a crucial role in biological and geological processes.",
        image: "https://upload.wikimedia.org/wikipedia/commons/5/56/Sulfur_crystals.jpg"
      },
      "water cycle": {
        text: "The water cycle describes how water evaporates, condenses in clouds, and returns to Earth as precipitation, powering ecosystems worldwide.",
        image: "https://upload.wikimedia.org/wikipedia/commons/thumb/3/3f/Water_cycle_diagram.svg/600px-Water_cycle_diagram.svg.png"
      },
      "metals and alloys": {
        text: "Metals are elements with high conductivity and malleability; alloys are mixtures of metals designed for specific properties.",
        image: "https://upload.wikimedia.org/wikipedia/commons/thumb/4/4f/Alloy_metals.png/600px-Alloy_metals.png"
      }
    };

    function generateSimulation() {
      const topic = document.getElementById('topicInput').value.trim().toLowerCase();
      const outputDiv = document.getElementById('output');

      if (simulations[topic]) {
        outputDiv.innerHTML = `
          <h2>${topic.charAt(0).toUpperCase() + topic.slice(1)}</h2>
          <p>${simulations[topic].text}</p>
          <img src="${simulations[topic].image}" alt="${topic} illustration" />
        `;
      } else if (topic === "") {
        outputDiv.innerHTML = "<p>Please enter a topic to generate simulation.</p>";
      } else {
        outputDiv.innerHTML = "<p>Simulation not available for this topic yet. Try 'nitrogen cycle', 'sulfur formation', 'water cycle', or 'metals and alloys'.</p>";
      }
    }
  </script>
